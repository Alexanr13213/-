#include <stdio.h>

struct thing
{
  int barcode;
  char name[15];
  int cost;
  float discount;
  float total;
  int num;
};
float a,b,c;
void addcheese()
    {   
        struct thing obj = { 1001,"Сыр", 63, 0.4 };
	    printf ("Введите количество:");
	    scanf ("%d", &obj.num);
	    printf ("\n %s %d Сведения о товаре:\n",obj.name, obj.barcode);
	    printf ("-------------------\n");
	    printf ("Цена %d\n", obj.cost);
	    printf ("Количество %d\n", obj.num);
	    printf ("Скидка %f\n", obj.discount);
	    float total = obj.num * obj.cost - obj.num * obj.cost * obj.discount;
	    obj.total=total;
	    printf ("Итог: %f\n", obj.total);
	    a=a+total;
	    c=c+obj.cost*obj.num;
    }
